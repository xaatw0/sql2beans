package sql2bean.dao;

import java.sql.ResultSet;
import java.sql.SQLException;

public class SelectSQLData implements ISQLSelect<SelectSQLData.Data>{

	// original SQL
	//   SELECT ID, FULL_NAME, MONEY FROM USER ORDER BY ID

	private final static String SQL_STATEMENT =
			"SELECT ID, FULL_NAME, MONEY FROM USER ORDER BY ID";

	@Override
	public String getSql() {
		return SQL_STATEMENT;
	}

	public class Data{
		// ID
		private Integer id;
		private void setId(Integer id){this.id = id;}
		public Integer getId(){ return id;}

		// FULL_NAME
		private String fullName;
		private void setFullName(String fullName){ this.fullName = fullName;}
		public String getFullName(){return fullName;}

		// MONEY
		private Integer money;
		private void setMoney(Integer money){this.money = money;}
		public Integer getMoney(){ return money;}

	}

	@Override
	public Data convert(ResultSet result) throws SQLException {

		Data data = new Data();

		data.setId(result.getInteger(1));
		data.setFullName(result.getString(2));
		data.setMoney(result.getInteger(3));

		return data;
	}
}
